1. Create `src/utils/escape.js`:
   ```js
   /**
    * Escape HTML entities to prevent XSS when interpolating user data in innerHTML.
    * @param {string} str - Raw string to escape
    * @returns {string} HTML-safe string
    */
   export function escapeHtml(str) {
     if (typeof str !== 'string') return String(str ?? '');
     return str
       .replace(/&/g, '&amp;')
       .replace(/</g, '&lt;')
       .replace(/>/g, '&gt;')
       .replace(/"/g, '&quot;')
       .replace(/'/g, '&#39;');
   }
   ```

2. Create `src/base/td-base-element.js` — the core base class (per D-01 through D-06):

   **Class structure (keep under 150 lines per pitfall #7):**

   ```js
   import { escapeHtml } from '../utils/escape.js';

   export class TdBaseElement extends HTMLElement {
     /** @abstract @returns {string[]} */
     static get observedAttributes() { return []; }

     /** Subclasses override to declare which observedAttributes are boolean */
     static get booleanAttributes() { return []; }

     constructor() {
       super();
       this._initialized = false;   // D-05: prevent duplicate render
       this._cleanups = [];          // D-03: track all disposables
     }

     connectedCallback() {
       if (!this._initialized) {
         this._initialized = true;
         this._setupProperties();    // D-02: auto getter/setter
         this._doRender();
       }
     }

     disconnectedCallback() {
       this._cleanups.forEach(fn => fn());
       this._cleanups = [];
     }

     // --- Rendering (D-01) ---

     /** @returns {string} HTML string. Override in subclass. */
     render() { return ''; }

     _doRender() {
       this.innerHTML = this.render();
       this.afterRender();
     }

     /** Hook for subclass to bind events after render. Called after every render. */
     afterRender() {}

     // --- Attribute/Property Sync (D-02, BASE-03) ---

     _setupProperties() {
       const booleans = new Set(this.constructor.booleanAttributes);
       for (const attr of this.constructor.observedAttributes) {
         if (attr in this) continue; // don't override existing
         const isBool = booleans.has(attr);
         // Convert kebab-case to camelCase for property name
         const prop = attr.replace(/-([a-z])/g, (_, c) => c.toUpperCase());
         Object.defineProperty(this, prop, {
           get: () => isBool ? this.hasAttribute(attr) : (this.getAttribute(attr) ?? ''),
           set: (val) => {
             if (isBool) {
               val ? this.setAttribute(attr, '') : this.removeAttribute(attr);
             } else {
               val == null ? this.removeAttribute(attr) : this.setAttribute(attr, val);
             }
           },
           configurable: true,
         });
       }
     }

     attributeChangedCallback(name, oldVal, newVal) {
       if (oldVal !== newVal && this._initialized) {
         this._doRender();
       }
     }

     // --- Cleanup Tracking (D-03, BASE-02) ---

     listen(target, event, handler, options) {
       target.addEventListener(event, handler, options);
       this._cleanups.push(() => target.removeEventListener(event, handler, options));
     }

     setTimeout(fn, ms) {
       const id = window.setTimeout(fn, ms);
       this._cleanups.push(() => window.clearTimeout(id));
       return id;
     }

     setInterval(fn, ms) {
       const id = window.setInterval(fn, ms);
       this._cleanups.push(() => window.clearInterval(id));
       return id;
     }

     // --- Event Emission (BASE-06) ---

     emit(name, detail = {}) {
       this.dispatchEvent(new CustomEvent(name, {
         bubbles: true,
         composed: true,
         detail,
       }));
     }

     // --- Utilities (BASE-04) ---

     escapeHtml(str) {
       return escapeHtml(str);
     }
   }
   ```

   Key decisions baked in:
   - `_initialized` flag per D-05 prevents duplicate render on DOM move
   - `listen()`, `setTimeout()`, `setInterval()` per D-03 track cleanups
   - `_setupProperties()` per D-02 auto-generates getter/setter for each observedAttribute
   - `booleanAttributes` static getter distinguishes boolean vs string attributes
   - `render()` returns HTML string, base class sets innerHTML per D-01
   - `emit()` helper for CustomEvent with bubbles + composed per BASE-06
   - `escapeHtml()` instance method delegates to utility per BASE-04
   - No auto-registration in base class — each component file handles its own per D-06
