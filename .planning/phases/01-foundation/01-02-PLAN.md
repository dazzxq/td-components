export class TdBaseElement extends HTMLElement {
  static get observedAttributes() { return []; }
  static get booleanAttributes() { return []; }
  render() { return ''; }           // override in subclass, return HTML string
  afterRender() {}                   // hook for binding events after render
  listen(target, event, handler, options) {}  // auto-cleanup event listener
  setTimeout(fn, ms) {}             // auto-cleanup timer
  setInterval(fn, ms) {}            // auto-cleanup interval
  emit(name, detail = {}) {}        // dispatch CustomEvent with bubbles+composed
  escapeHtml(str) {}                // XSS-safe string escaping
}

     render() {
       return `<div>${this.escapeHtml(this.getAttribute('label') || '')}</div>`;

   }
